# include <fstream>
#include <string>
#include <iostream>
const int one=96679,two=99779;
using namespace std;

int sort(int a[one],int key,int min, int max)
{
if (min>max) return -1;
int sr = min+(max-min)/2;
if(key==a[sr]) return sr;
if(key>a[sr]) return sort(a,key,sr+1,max);
else return sort(a,key,min,sr-1);
}

 
int main()
{setlocale(LC_ALL,"Rus");
    fstream fout, fout1;
    fout.open("out.txt");
	ofstream fly("output.txt");
	fout1.open("out1.txt");
    int a[one],b[two],n=0,l=0;
 
    while (!fout.eof())
    {
        fout>>a[n];
		n++;
    }

	while (!fout1.eof())
    {
        fout1>>b[l];
		l++;
    }
	int key,min=0,max=one;
	for(int k=0;k<two;k++)
{
	key=b[k];
	key=sort(a,key,min,max);
    if (key==-1)
      fly<<key<<" ";
    else
      fly<<key+1<<" ";
}
	fly.close();
    fout.close();
	fout1.close();
    system("pause");
    
 
    }
