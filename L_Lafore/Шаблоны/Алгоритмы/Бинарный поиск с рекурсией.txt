#include<iostream>
using namespace std;
int sort(int a[100],int key,int min, int max)
{if (min>max) return -1;
int sr = min+(max-min)/2;
if(key==a[sr]) return sr;
if(key>a[sr]) return sort(a,key,sr+1,max);
else return sort(a,key,min,sr-1);
}
int main()
{setlocale(LC_ALL,"Rus");
int a[100],key,min=0,max=100;
for(int j=900;j<1000;j++)
	a[j-900]=j;
cout<<"Введите число --> ";
cin>>key;
key=sort(a,key,min,max);
if (key==-1)
cout<<"Ошибка! ";
else
cout<<key;
cin.get();
cin.get();
return 0;
}
